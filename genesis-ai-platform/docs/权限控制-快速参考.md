# 权限控制 - 快速参考

## 一句话总结

**在注册时添加 `*_permissions` 参数即可，零额外代码！**

## 快速示例

### 1. 不需要权限（只需登录）

```python
crud_factory.register(
    model=Tag,
    prefix="/tags",
    tags=["tags"]
)
```

### 2. 所有操作需要相同权限

```python
crud_factory.register(
    model=Document,
    prefix="/documents",
    tags=["documents"],
    list_permissions=["doc:read", "admin"],
    get_permissions=["doc:read", "admin"],
    create_permissions=["doc:read", "admin"],
    update_permissions=["doc:read", "admin"],
    delete_permissions=["doc:read", "admin"]
)
```

### 3. 不同操作不同权限（推荐）

```python
crud_factory.register(
    model=KnowledgeBase,
    prefix="/knowledge-bases",
    tags=["knowledge-bases"],
    list_permissions=["kb:read", "admin"],      # 读
    get_permissions=["kb:read", "admin"],       # 读
    create_permissions=["kb:write", "admin"],   # 写
    update_permissions=["kb:write", "admin"],   # 写
    delete_permissions=["kb:delete", "admin"]   # 删除
)
```

### 4. 只读资源

```python
crud_factory.register(
    model=AuditLog,
    prefix="/audit-logs",
    tags=["audit-logs"],
    list_permissions=["audit:read", "admin"],
    get_permissions=["audit:read", "admin"],
    enable_create=False,
    enable_update=False,
    enable_delete=False
)
```

### 5. 管理员专属

```python
crud_factory.register(
    model=SystemConfig,
    prefix="/system-configs",
    tags=["system-configs"],
    list_permissions=["admin"],
    get_permissions=["admin"],
    create_permissions=["admin"],
    update_permissions=["admin"],
    delete_permissions=["admin"]
)
```

### 6. 公开读，限制写

```python
crud_factory.register(
    model=Tag,
    prefix="/tags",
    tags=["tags"],
    list_permissions=None,                      # 只需登录
    get_permissions=None,                       # 只需登录
    create_permissions=["tag:write", "admin"],  # 需要权限
    update_permissions=["tag:write", "admin"],  # 需要权限
    delete_permissions=["tag:delete", "admin"]  # 需要权限
)
```

## 权限参数

| 参数 | 说明 | 默认值 |
|------|------|--------|
| `list_permissions` | 列表操作所需权限 | `None`（只需登录） |
| `get_permissions` | 获取单个资源所需权限 | `None`（只需登录） |
| `create_permissions` | 创建操作所需权限 | `None`（只需登录） |
| `update_permissions` | 更新操作所需权限 | `None`（只需登录） |
| `delete_permissions` | 删除操作所需权限 | `None`（只需登录） |

## 权限逻辑

- **`None`**：只需要登录用户
- **`["perm1", "perm2"]`**：用户需要拥有 `perm1` **或** `perm2`（OR 逻辑）

## 权限命名建议

```python
# 格式：资源:操作
"kb:read"       # 知识库读取
"kb:write"      # 知识库写入
"kb:delete"     # 知识库删除
"doc:upload"    # 文档上传
"user:manage"   # 用户管理

# 特殊权限
"admin"         # 管理员
"superuser"     # 超级用户
```

## 常见场景

### 场景 1：普通业务表

```python
# 大部分表：读写分离
crud_factory.register(
    model=YourModel,
    prefix="/your-resources",
    tags=["your-resources"],
    list_permissions=["resource:read", "admin"],
    get_permissions=["resource:read", "admin"],
    create_permissions=["resource:write", "admin"],
    update_permissions=["resource:write", "admin"],
    delete_permissions=["resource:delete", "admin"]
)
```

### 场景 2：公开资源

```python
# 所有人可以查看，只有管理员可以修改
crud_factory.register(
    model=PublicResource,
    prefix="/public-resources",
    tags=["public-resources"],
    # 不指定权限，只需登录
)
```

### 场景 3：敏感资源

```python
# 只有管理员可以访问
crud_factory.register(
    model=SensitiveData,
    prefix="/sensitive-data",
    tags=["sensitive-data"],
    list_permissions=["admin"],
    get_permissions=["admin"],
    create_permissions=["admin"],
    update_permissions=["admin"],
    delete_permissions=["admin"]
)
```

## 完整示例

```python
# api/crud_registry.py
from core.crud_factory import crud_factory
from models.knowledge_base import KnowledgeBase
from models.document import Document
from models.tag import Tag
from models.audit_log import AuditLog

def register_all_crud():
    # 知识库：读写分离
    crud_factory.register(
        model=KnowledgeBase,
        prefix="/knowledge-bases",
        tags=["knowledge-bases"],
        list_permissions=["kb:read", "kb:write", "admin"],
        get_permissions=["kb:read", "kb:write", "admin"],
        create_permissions=["kb:write", "admin"],
        update_permissions=["kb:write", "admin"],
        delete_permissions=["kb:delete", "admin"]
    )
    
    # 文档：细粒度权限
    crud_factory.register(
        model=Document,
        prefix="/documents",
        tags=["documents"],
        list_permissions=["doc:read", "doc:write", "admin"],
        get_permissions=["doc:read", "doc:write", "admin"],
        create_permissions=["doc:write", "doc:upload", "admin"],
        update_permissions=["doc:write", "admin"],
        delete_permissions=["doc:delete", "admin"]
    )
    
    # 标签：公开读，限制写
    crud_factory.register(
        model=Tag,
        prefix="/tags",
        tags=["tags"],
        list_permissions=None,
        get_permissions=None,
        create_permissions=["tag:write", "admin"],
        update_permissions=["tag:write", "admin"],
        delete_permissions=["tag:delete", "admin"]
    )
    
    # 审计日志：只读
    crud_factory.register(
        model=AuditLog,
        prefix="/audit-logs",
        tags=["audit-logs"],
        list_permissions=["audit:read", "admin"],
        get_permissions=["audit:read", "admin"],
        enable_create=False,
        enable_update=False,
        enable_delete=False
    )
```
